JSON ফাইল স্ট্রাকচার নির্দেশিকা

আপনার JSON ফাইলটি অবশ্যই নিচের কাঠামো অনুসরণ করে তৈরি করতে হবে:

{
  "metadata": {
    "chapter": "অধ্যায়ের নাম",
    "class": "শ্রেণী (যেমন: HSC)",
    "subject": "বিষয়ের নাম",
    "totalMarks": 100,
    "duration": "0 minutes"
  },
  "questions": [
    {
      "id": 1,
      "question": "MCQ প্রশ্ন টেক্সট",
      "compact": true, // অপশনগুলো পাশাপাশি দেখাতে 
      "options": [
        "অপশন ক",
        "অপশন খ",
        "img:https://example.com/image.png", // ছবির অপশন হলে img: প্রোটোকল 
        "অপশন ঘ"
      ],
      "answer": 0, // সঠিক উত্তরের ইনডেক্স (0 থেকে শুরু)
      "explanation": "ব্যাখ্যা টেক্সট",
      "image": "https://example.com/question_image.png" // ঐচ্ছিক
    },
    {
      "id": 2,
      "type": "cq", // সৃজনশীল প্রশ্ন (CQ) এর জন্য
      "stem": {
        "text": "উদ্দীপক টেক্সট",
        "image": { // ঐচ্ছিক
          "src": "https://example.com/stem_image.png",
          "alt": "Image alt text"
        }
      },
      "questions": [
        {
          "qid": "গ",
          "question": "গ নং প্রশ্ন",
          "answer": "উত্তর",
          "explanation": "ব্যাখ্যা"
        }
      ]
    }
  ]
}
